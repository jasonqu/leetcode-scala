def findMinRecur(l: Array[Int], start: Int, end: Int): Int = {
  if (start == end - 1) l(end)
  else {
    val mid = (start + end) / 2
    if (l(start) == l(mid))
      findMinRecur(l, start + 1, mid)
    else if (l(start) > l(mid))
      findMinRecur(l, start, mid)
